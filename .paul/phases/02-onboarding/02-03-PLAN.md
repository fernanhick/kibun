    **src/store/onboardingGateStore.ts — new file**
    Persisted (AsyncStorage) — survives app kill/restart. This is how the gate "remembers" onboarding was completed.
    // AUDIT M3: must handle AsyncStorage rehydration lag. On first render, Zustand returns the
    // default state (complete: false) while AsyncStorage is still loading. Without a hydration
    // guard, the gate incorrectly redirects returning users to onboarding before rehydration.
    // Fix: _hasHydrated transient flag + onRehydrateStorage callback + partialize to exclude it.
    ```ts
    import { create } from 'zustand';
    import { persist, createJSONStorage } from 'zustand/middleware';
    import AsyncStorage from '@react-native-async-storage/async-storage';

    interface OnboardingGateState {
      complete: boolean;
      _hasHydrated: boolean;
      setComplete: () => void;
      setHasHydrated: (value: boolean) => void;
    }

    export const useOnboardingGateStore = create<OnboardingGateState>()(
      persist(
        (set) => ({
          complete: false,
          _hasHydrated: false,
          setComplete: () => set({ complete: true }),
          setHasHydrated: (value) => set({ _hasHydrated: value }),
        }),
        {
          name: 'kibun-onboarding-gate',
          storage: createJSONStorage(() => AsyncStorage),
          // Only persist 'complete' — _hasHydrated is transient (always starts false on launch)
          partialize: (state) => ({ complete: state.complete }),
          onRehydrateStorage: () => (state) => {
            state?.setHasHydrated(true);
          },
        }
      )
    );
    ```
